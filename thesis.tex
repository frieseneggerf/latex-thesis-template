%==== Function \formatDNA{} takes string and separates it into groups of 3 characters======
%Useful for making dna sequences readable & look good (ofc no relation to biological readingframe)
\makeatletter%
\newcounter{@count}%
\newcommand{\formatDNA}[1]{%
	\setcounter{@count}{0}%
	\@tfor\base:=#1\do{%
		\base% 
		\stepcounter{@count}%
		\ifnum\value{@count}=3
		\space%
		\setcounter{@count}{0}%
		\fi%
	}%
}%
\makeatother%
